package main

import (
    "net/http"
    "runtime"
)

func main() {
    http.HandleFunc("/", handler)
    http.ListenAndServe(":8080", nil)
}

func handler(w http.ResponseWriter, r *http.Request) {
    if runtime.GOOS == "darwin" {
        w.Write([]byte("hello from darwin"))
    } else {
        w.Write([]byte("hello from linux"))
    }
}
